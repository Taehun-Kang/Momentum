// components/common/LazyImage.js
export class LazyImage extends Component {
  constructor(props) {
    super({ props, tagName: "img" });
    this.observer = null;
  }

  connectedCallback() {
    this.setupIntersectionObserver();
  }

  setupIntersectionObserver() {
    const options = {
      root: null,
      rootMargin: "50px", // 50px ë¯¸ë¦¬ ë¡œë“œ
      threshold: 0.01,
    };

    this.observer = new IntersectionObserver((entries) => {
      entries.forEach((entry) => {
        if (entry.isIntersecting) {
          this.loadImage();
          this.observer.unobserve(entry.target);
        }
      });
    }, options);

    this.observer.observe(this.el);
  }

  loadImage() {
    const src = this.props.src;
    const img = new Image();

    img.onload = () => {
      this.el.src = src;
      this.el.classList.add("loaded");
    };

    // WebP ì§€ì› í™•ì¸
    if (this.supportsWebP()) {
      img.src = src.replace(/\.(jpg|png)$/, ".webp");
    } else {
      img.src = src;
    }
  }

  supportsWebP() {
    const canvas = document.createElement("canvas");
    canvas.width = 1;
    canvas.height = 1;
    return canvas.toDataURL("image/webp").indexOf("image/webp") === 0;
  }
}

export function virtualScroll(container, items, itemHeight, renderItem) {
  const visibleCount = Math.ceil(container.clientHeight / itemHeight);
  const totalHeight = items.length * itemHeight;
  let scrollTop = 0;

  const content = document.createElement("div");
  content.style.height = `${totalHeight}px`;
  content.style.position = "relative";

  const renderVisibleItems = () => {
    const startIndex = Math.floor(scrollTop / itemHeight);
    const endIndex = Math.min(startIndex + visibleCount + 1, items.length);

    content.innerHTML = "";

    for (let i = startIndex; i < endIndex; i++) {
      const item = renderItem(items[i], i);
      item.style.position = "absolute";
      item.style.top = `${i * itemHeight}px`;
      content.appendChild(item);
    }
  };

  container.addEventListener("scroll", () => {
    scrollTop = container.scrollTop;
    requestAnimationFrame(renderVisibleItems);
  });

  container.appendChild(content);
  renderVisibleItems();
}

      'a[href], button, textarea, input[type="text"], ' +
